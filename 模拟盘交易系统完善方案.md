## 一、交易记录关联化（订单配对管理）

### 1.1 问题分析

当前 `paper_trades` 表每笔成交记录独立存储，买入卖出订单没有关联关系，导致：
- 无法追踪完整交易生命周期
- 难以计算单笔交易的真正盈亏
- 持仓分析不准确

### 1.2 数据模型设计

### 1.3 核心逻辑实现

**配对规则**：
1. **LONG（多头）**：先买入(BUY) → 后卖出(SELL) 配对
2. **SHORT（空头）**：先卖出(SUY) → 后买入(BUY) 配对
3. **FIFO原则**：按时间顺序，先进先出配对

**实现要点**：

```python
# backend/app/services/trade_pair_service.py

from datetime import datetime
from decimal import Decimal
from typing import List, Optional, Dict
import uuid

class TradePairService:
    
    async def on_trade_filled(self, trade: PaperTrade) -> Optional[TradePair]:
        """订单成交时调用，自动配对"""
        
        # 查找同一symbol的未平仓配对
        existing_pair = await self._find_open_pair(trade.symbol, trade.side)
        
        if trade.side == "BUY":
            if existing_pair and existing_pair.side == "SHORT":
                # 平空仓：配对平仓
                return await self._close_pair(existing_pair, trade)
            else:
                # 开多仓：创建新配对
                return await self._create_pair(trade)
        else:  # SELL
            if existing_pair and existing_pair.side == "LONG":
                # 平多仓：配对平仓
                return await self._close_pair(existing_pair, trade)
            else:
                # 开空仓：创建新配对
                return await self._create_pair(trade)
    
    async def _create_pair(self, trade: PaperTrade) -> TradePair:
        """创建新交易对（开仓）"""
        pair = TradePair(
            pair_id=str(uuid.uuid4()),
            symbol=trade.symbol,
            entry_trade_id=trade.id,
            exit_trade_id=None,
            entry_time=trade.created_at,
            entry_price=trade.price,
            quantity=trade.quantity,
            side="LONG" if trade.side == "BUY" else "SHORT",
            status="OPEN",
            holding_costs=trade.fee
        )
        return await self._save_pair(pair)
    
    async def _close_pair(self, pair: TradePair, exit_trade: PaperTrade) -> TradePair:
        """平仓：更新配对信息"""
        pair.exit_trade_id = exit_trade.id
        pair.exit_time = exit_trade.created_at
        pair.exit_price = exit_trade.price
        pair.status = "CLOSED"
        pair.holding_costs += exit_trade.fee
        
        # 计算盈亏
        if pair.side == "LONG":
            pnl = (pair.exit_price - pair.entry_price) * pair.quantity - pair.holding_costs
            pnl_pct = (pair.exit_price / pair.entry_price - 1) * 100
        else:  # SHORT
            pnl = (pair.entry_price - pair.exit_price) * pair.quantity - pair.holding_costs
            pnl_pct = (pair.entry_price / pair.exit_price - 1) * 100
        
        pair.pnl = pnl
        pair.pnl_pct = pnl_pct
        
        # 计算持仓时间
        delta = pair.exit_time - pair.entry_time
        pair.holding_hours = delta.total_seconds() / 3600
        
        return await self._save_pair(pair)
    
    async def get_pair_trades(self, pair_id: str) -> Dict:
        """获取交易对的完整流水"""
        pair = await self._get_pair(pair_id)
        entry_trade = await self._get_trade(pair.entry_trade_id)
        exit_trade = await self._get_trade(pair.exit_trade_id) if pair.exit_trade_id else None
        
        return {
            "pair_id": pair.pair_id,
            "symbol": pair.symbol,
            "status": pair.status,
            "entry": {
                "trade_id": entry_trade.id,
                "time": entry_trade.created_at.isoformat(),
                "side": pair.side,
                "price": float(entry_trade.price),
                "quantity": float(entry_trade.quantity),
                "fee": float(entry_trade.fee)
            },
            "exit": {
                "trade_id": exit_trade.id,
                "time": exit_trade.created_at.isoformat(),
                "price": float(exit_trade.price),
                "quantity": float(exit_trade.quantity),
                "fee": float(exit_trade.fee)
            } if exit_trade else None,
            "pnl": float(pair.pnl) if pair.pnl else None,
            "pnl_pct": float(pair.pnl_pct) if pair.pnl_pct else None,
            "holding_hours": float(pair.holding_hours) if pair.holding_hours else None,
            "holding_costs": float(pair.holding_costs)
        }
```

## 二、完整性能指标体系

### 2.1 指标定义

| 指标 | 计算方法 | 说明 |
|------|---------|------|
| **收益率** | (期末权益 - 期初权益) / 期初权益 | 总收益率 |
| **年化收益率** | 日收益率 * 252 | 年化后收益率 |
| **夏普比率** | (年化收益率 - 无风险利率) / 年化波动率 | 风险调整收益 |
| **最大回撤** | max(Peak - Portfolio) / Peak | 历史最大亏损 |
| **胜率** | 盈利交易数 / 总交易数 | 盈利概率 |
| **盈亏比** | 平均盈利 / 平均亏损 | 盈亏平衡比 |
| **卡玛比率** | 年化收益率 / 最大回撤 | 收益/回撤比 |
| **_sortino比率** | (年化收益率 - 目标收益) / 下行波动率 | 只考虑下行风险 |

### 2.3 性能计算服务

```python
# backend/app/services/performance_service.py

from datetime import datetime, timedelta
from decimal import Decimal
from typing import Dict, List, Optional
from collections import defaultdict
import numpy as np

class PerformanceService:
    RISK_FREE_RATE = 0.03  # 3% 无风险利率
    
    async def calculate_metrics(
        self, 
        start_date: datetime,
        end_date: datetime,
        initial_capital: Decimal
    ) -> Dict:
        """计算完整性能指标"""
        
        # 1. 获取权益曲线
        equity_curve = await self._get_equity_curve(start_date, end_date)
        
        # 2. 获取已平仓交易
        closed_pairs = await self._get_closed_trade_pairs(start_date, end_date)
        
        # 3. 计算各项指标
        returns = self._calculate_returns(equity_curve)
        
        final_equity = equity_curve[-1]['total_equity'] if equity_curve else initial_capital
        total_return = (float(final_equity) / float(initial_capital) - 1) * 100
        
        # 基础指标
        metrics = {
            'initial_capital': float(initial_capital),
            'final_equity': float(final_equity),
            'total_return': total_return,
            'total_trades': len(closed_pairs),
            'winning_trades': sum(1 for p in closed_pairs if p.pnl and p.pnl > 0),
            'losing_trades': sum(1 for p in closed_pairs if p.pnl and p.pnl < 0),
        }
        
        # 胜率 & 盈亏比
        metrics['win_rate'] = (metrics['winning_trades'] / metrics['total_trades'] * 100) if metrics['total_trades'] > 0 else 0
        
        total_profit = sum(float(p.pnl) for p in closed_pairs if p.pnl and p.pnl > 0)
        total_loss = abs(sum(float(p.pnl) for p in closed_pairs if p.pnl and p.pnl < 0))
        metrics['profit_factor'] = total_profit / total_loss if total_loss > 0 else float('inf')
        metrics['avg_profit'] = total_profit / metrics['winning_trades'] if metrics['winning_trades'] > 0 else 0
        metrics['avg_loss'] = total_loss / metrics['losing_trades'] if metrics['losing_trades'] > 0 else 0
        
        # 风险指标
        max_dd, max_dd_pct = self._calculate_max_drawdown(equity_curve)
        metrics['max_drawdown'] = float(max_dd)
        metrics['max_drawdown_pct'] = max_dd_pct
        
        volatility = self._calculate_volatility(returns)
        metrics['volatility'] = volatility
        
        # 收益率指标
        days = (end_date - start_date).days
        years = days / 365
        annualized_return = ((float(final_equity) / float(initial_capital)) ** (1/years) - 1) * 100 if years > 0 else 0
        metrics['annualized_return'] = annualized_return
        
        # 夏普比率
        if volatility > 0:
            metrics['sharpe_ratio'] = (annualized_return - self.RISK_FREE_RATE * 100) / (volatility * 100)
        else:
            metrics['sharpe_ratio'] = 0
        
        # Sortino比率（下行波动率）
        downside_returns = [r for r in returns if r < 0]
        downside_vol = np.std(downside_returns) * np.sqrt(252) if downside_returns else 0
        if downside_vol > 0:
            metrics['sortino_ratio'] = (annualized_return - self.RISK_FREE_RATE * 100) / (downside_vol * 100)
        
        # 卡玛比率
        if max_dd_pct > 0:
            metrics['calmar_ratio'] = annualized_return / max_dd_pct
        
        # 持仓时间统计
        holding_hours = [float(p.holding_hours) for p in closed_pairs if p.holding_hours]
        metrics['avg_holding_hours'] = np.mean(holding_hours) if holding_hours else 0
        
        # 连续盈亏
        metrics['max_consecutive_wins'] = self._max_consecutive(closed_pairs, True)
        metrics['max_consecutive_losses'] = self._max_consecutive(closed_pairs, False)
        
        return metrics
    
    def _calculate_returns(self, equity_curve: List[Dict]) -> List[float]:
        """计算收益率序列"""
        if len(equity_curve) < 2:
            return []
        
        returns = []
        for i in range(1, len(equity_curve)):
            prev = equity_curve[i-1]['total_equity']
            curr = equity_curve[i]['total_equity']
            if prev > 0:
                ret = (curr - prev) / prev
                returns.append(ret)
        return returns
    
    def _calculate_max_drawdown(self, equity_curve: List[Dict]) -> tuple:
        """计算最大回撤"""
        if not equity_curve:
            return 0, 0
        
        peak = equity_curve[0]['total_equity']
        max_dd = 0
        max_dd_pct = 0
        
        for point in equity_curve:
            equity = point['total_equity']
            if equity > peak:
                peak = equity
            
            dd = peak - equity
            dd_pct = (dd / peak * 100) if peak > 0 else 0
            
            if dd > max_dd:
                max_dd = dd
                max_dd_pct = dd_pct
        
        return max_dd, max_dd_pct
    
    def _calculate_volatility(self, returns: List[float]) -> float:
        """计算年化波动率"""
        if not returns:
            return 0
        return np.std(returns) * np.sqrt(252)
    
    def _max_consecutive(self, pairs: List, winning: bool) -> int:
        """计算最大连续盈利/亏损"""
        max_streak = current_streak = 0
        
        for pair in pairs:
            if pair.pnl is None:
                continue
            
            is_win = pair.pnl > 0
            if is_win == winning:
                current_streak += 1
                max_streak = max(max_streak, current_streak)
            else:
                current_streak = 0
        
        return max_streak
```

```python
        # 获取当前余额
        balance_data = await paper_trading_service.get_balance()

        prev = await session.execute(
            "SELECT * FROM equity_snapshots ORDER BY timestamp DESC LIMIT 1"
        )
        prev_snapshot = prev.fetchone()
        
        if prev_snapshot:
            daily_pnl = total_equity - Decimal(str(prev_snapshot.total_equity))
            daily_return = (daily_pnl / Decimal(str(prev_snapshot.total_equity))) * 100 if prev_snapshot.total_equity > 0 else 0
        else:
            daily_pnl = 0
            daily_return = 0
        
        # 计算当前回撤
        peak_result = await session.execute(
            "SELECT MAX(total_equity) as peak FROM equity_snapshots"
        )
        peak = peak_result.scalar() or total_equity
        drawdown = ((peak - total_equity) / peak * 100) if peak > 0 else 0
        
        snapshot = EquitySnapshot(
            timestamp=datetime.utcnow(),
            total_equity=total_equity,
            cash_balance=cash_balance,
            position_value=position_value,
            daily_pnl=daily_pnl,
            daily_return=daily_return,
            drawdown=drawdown
        )
        session.add(snapshot)
        await session.commit()
```

## 三、实时持仓分析

### 3.1 增强持仓数据结构

```python
# backend/app/services/position_analysis_service.py

from datetime import datetime
from typing import Dict, List
from decimal import Decimal

class PositionAnalysisService:
    
    async def get_position_analytics(self, symbol: str, current_price: Decimal) -> Dict:
        """获取单币种持仓分析"""
        
        # 获取当前持仓
        position = await self._get_position(symbol)
        if not position or position.quantity == 0:
            return None
        
        # 获取该币种所有交易对
        pairs = await self._get_trade_pairs(symbol)
        closed_pairs = [p for p in pairs if p.status == "CLOSED"]
        open_pairs = [p for p in pairs if p.status == "OPEN"]
        
        # 计算各项指标
        qty = Decimal(str(position.quantity))
        avg_price = Decimal(str(position.avg_price))
        
        # 浮盈浮亏
        unrealized_pnl = (current_price - avg_price) * qty
        unrealized_pnl_pct = ((current_price / avg_price - 1) * 100) if avg_price > 0 else 0
        
        # 持仓时间分析
        if open_pairs:
            entry_times = [p.entry_time for p in open_pairs]
            avg_entry_time = min(entry_times)  # 最早入场时间
            holding_seconds = (datetime.utcnow() - avg_entry_time).total_seconds()
            holding_hours = holding_seconds / 3600
        else:
            holding_hours = 0
        
        # 风险暴露
        total_value = current_price * abs(qty)
        
        # 历史交易统计
        total_trades = len(closed_pairs)
        winning_trades = sum(1 for p in closed_pairs if p.pnl and p.pnl > 0)
        win_rate = (winning_trades / total_trades * 100) if total_trades > 0 else 0
        
        avg_pnl = sum(float(p.pnl) for p in closed_pairs if p.pnl) / total_trades if total_trades > 0 else 0
        
        return {
            "symbol": symbol,
            "side": "LONG" if qty > 0 else "SHORT",
            "quantity": float(qty),
            "avg_price": float(avg_price),
            "current_price": float(current_price),
            
            # 浮盈浮亏
            "unrealized_pnl": float(unrealized_pnl),
            "unrealized_pnl_pct": float(unrealized_pnl_pct),
            
            # 持仓分析
            "holding_hours": round(holding_hours, 2),
            "open_position_count": len(open_pairs),
            "entry_time": avg_entry_time.isoformat() if open_pairs else None,
            
            # 风险暴露
            "position_value": float(total_value),
            "position_value_pct": 0,  # 需结合总资产计算
            
            # 历史统计
            "total_trades": total_trades,
            "winning_trades": winning_trades,
            "win_rate": round(win_rate, 2),
            "avg_pnl": round(avg_pnl, 2),
            
            # 盈亏来源
            "realized_pnl": sum(float(p.pnl) for p in closed_pairs if p.pnl),
        }
    
    async def get_portfolio_analytics(
        self, 
        current_prices: Dict[str, float]
    ) -> Dict:
        """获取组合层面分析"""
        
        positions = await paper_trading_service.get_positions(current_prices)
        
        # 汇总数据
        total_position_value = sum(p['quantity'] * p['mark_price'] for p in positions)
        total_unrealized_pnl = sum(p['pnl'] for p in positions)
        
        # 获取账户余额
        balance = await paper_trading_service.get_balance()
        cash = balance['total_balance']
        
        total_equity = cash + total_position_value
        
        # 资产配置
        asset_allocation = []
        for pos in positions:
            value = pos['quantity'] * pos['mark_price']
            allocation_pct = (value / total_equity * 100) if total_equity > 0 else 0
            asset_allocation.append({
                "symbol": pos['symbol'],
                "value": value,
                "allocation_pct": round(allocation_pct, 2),
                "side": pos['side']
            })
        
        # 风险指标
        exposure_by_side = {
            "LONG": sum(p['quantity'] * p['mark_price'] for p in positions if p.get('side') == 'LONG'),
            "SHORT": sum(p['quantity'] * p['mark_price'] for p in positions if p.get('side') == 'SHORT'),
        }
        
        return {
            "total_equity": total_equity,
            "cash": cash,
            "position_value": total_position_value,
            "cash_pct": round(cash / total_equity * 100, 2) if total_equity > 0 else 100,
            "total_unrealized_pnl": total_unrealized_pnl,
            
            "asset_allocation": asset_allocation,
            
            "exposure": {
                "long": exposure_by_side['LONG'],
                "short": exposure_by_side['SHORT'],
                "net_exposure": exposure_by_side['LONG'] - exposure_by_side['SHORT'],
                "gross_exposure": exposure_by_side['LONG'] + exposure_by_side['SHORT'],
                "leverage": (exposure_by_side['LONG'] + exposure_by_side['SHORT']) / total_equity if total_equity > 0 else 0
```

## 六、量化核心补充：指标严谨性与回测-实盘一致性

### 6.1 关键指标计算的正确方法论

#### 6.1.1 夏普比率 (Sharpe Ratio)
**标准公式**：
\[ Sharpe = \frac{R_p - R_f}{\sigma_p} \]
- \( R_p \): 组合年化收益率（复合年化，非简单年化）
- \( R_f \): 无风险利率（建议使用同期国债收益率，默认3%）
- \( \sigma_p \): 组合年化波动率（日收益率标准差 × √252）

**实现修正**：
```python
def calculate_sharpe_ratio(returns: List[float], risk_free_rate: float = 0.03) -> float:
    """
    计算夏普比率
    returns: 日收益率序列（小数，如0.01表示1%）
    """
    if len(returns) < 2:
        return 0.0
    
    # 超额收益
    excess_returns = np.array(returns) - (risk_free_rate / 252)
    
    # 年化波动率
    volatility = np.std(excess_returns) * np.sqrt(252)
    
    if volatility == 0:
        return 0.0
    
    # 年化超额收益
    annualized_excess_return = np.mean(excess_returns) * 252
    
    return annualized_excess_return / volatility
```

#### 6.1.2 卡玛比率 (Calmar Ratio)
**标准公式**：
\[ Calmar = \frac{年化收益率}{最大回撤（绝对值）} \]
- 最大回撤使用**百分比**而非绝对值金额
- 建议用**近3年**或**全部历史**数据计算

#### 6.1.3 索提诺比率 (Sortino Ratio)
**标准公式**：
\[ Sortino = \frac{R_p - R_f}{\sigma_d} \]
- \( \sigma_d \): 下行波动率（仅考虑负收益的标准差）
- 下行偏离目标：可设为0或无风险利率

#### 6.1.4 最大回撤 (Max Drawdown)
**标准计算方法**：
```python
def calculate_max_drawdown(equity_curve: List[float]) -> tuple:
    """
    计算最大回撤和最大回撤百分比
    返回: (回撤金额, 回撤百分比, 峰值索引, 谷值索引)
    """
    peak = equity_curve[0]
    max_dd = 0.0
    max_dd_pct = 0.0
    peak_idx = 0
    trough_idx = 0
    current_peak_idx = 0
    
    for i, equity in enumerate(equity_curve):
        if equity > peak:
            peak = equity
            current_peak_idx = i
        
        dd = peak - equity
        dd_pct = dd / peak if peak > 0 else 0
        
        if dd > max_dd:
            max_dd = dd
            max_dd_pct = dd_pct
            peak_idx = current_peak_idx
            trough_idx = i
    
    return max_dd, max_dd_pct * 100, peak_idx, trough_idx
```

### 6.2 交易配对逻辑完善（复杂场景处理）

#### 6.2.1 支持部分平仓
当前方案假设每次都是全量配对，实际需要支持**部分平仓**：

```python
async def _close_pair_partial(self, pair: TradePair, exit_trade: PaperTrade, close_qty: Decimal) -> List[TradePair]:
    """
    部分平仓：将原配对拆分为"已平仓部分"和"剩余持仓部分"
    """
    # 1. 创建新的配对记录已平仓部分
    closed_pair = await self._create_pair_from_existing(pair, close_qty)
    closed_pair = await self._close_pair(closed_pair, exit_trade)
    
    # 2. 更新原配对为剩余数量
    pair.quantity -= close_qty
    
    return [closed_pair, pair]
```

#### 6.2.2 支持多笔开仓合并平仓
```python
async def _close_multiple_pairs(self, pairs: List[TradePair], exit_trade: PaperTrade) -> List[TradePair]:
    """
    一笔平仓订单平掉多笔开仓（按FIFO顺序）
    """
    remaining_qty = exit_trade.quantity
    closed_pairs = []
    
    for pair in sorted(pairs, key=lambda p: p.entry_time):
        if remaining_qty <= 0:
            break
        
        close_qty = min(pair.quantity, remaining_qty)
        
        if close_qty == pair.quantity:
            # 全平
            closed_pair = await self._close_pair(pair, exit_trade)
            closed_pairs.append(closed_pair)
        else:
            # 部分平
            results = await self._close_pair_partial(pair, exit_trade, close_qty)
            closed_pairs.extend(results)
        
        remaining_qty -= close_qty
    
    return closed_pairs
```

### 6.3 回测-实盘一致性保障

#### 6.3.1 滑点与手续费模型
**必须与回测保持一致**：
- 买入：`成交价 = 卖一价 × (1 + 滑点)`
- 卖出：`成交价 = 买一价 × (1 - 滑点)`
- 手续费率：与交易所设置完全相同（Maker/Taker区分）

#### 6.3.2 资金费用处理（合约特有）
```python
async def calculate_funding_fees(self, pairs: List[TradePair]) -> Decimal:
    """
    计算持仓期间资金费用
    币安合约：每8小时结算一次
    """
    total_fee = Decimal('0')
    
    for pair in pairs:
        if pair.status != "OPEN":
            continue

            position_value = pair.quantity * pair.entry_price  # 简化，实际应按标记价格
            fee = position_value * Decimal(str(rate.rate))
```

### 6.4 风险暴露深度分析

#### 6.4.1 VaR（风险价值）计算
```python
def calculate_var(returns: List[float], confidence_level: float = 0.95) -> float:
    """
    计算历史模拟法VaR
    returns: 日收益率序列
    """
    if len(returns) < 20:
        return 0.0
    
    sorted_returns = np.sort(returns)
    var_index = int((1 - confidence_level) * len(sorted_returns))
    
    return -sorted_returns[var_index] * 100  # 转为百分比正数
```

#### 6.4.2 持仓集中度
```python
def calculate_concentration(position_values: List[float]) -> float:
    """
    计算赫芬达尔-赫希曼指数(HHI)衡量集中度
    0-10000，越高越集中
    """
    total = sum(position_values)
    if total == 0:
        return 0
    
    weights = [(v / total) ** 2 for v in position_values]
    return sum(weights) * 10000
```

```tsx
interface EquityCurveChartProps {

  showDrawdown = true,
  initialCapital = 100000,

  const peakPoints = useMemo(() => {

    return data.map((point, index) => {
      if (point.total_equity > peak) {
        peak = point.total_equity;
        return { ...point, isPeak: true };
      }
      return { ...point, isPeak: false };
    });
  }, [data]);

  // 自定义Tooltip
  const CustomTooltip = ({ active, payload, label }: any) => {
    if (active && payload && payload.length) {
      const point = payload[0].payload;
      const returnPct = ((point.total_equity / initialCapital) - 1) * 100;

          <p className="text-slate-400 text-sm mb-2">{formatDate(label)}</p>

            现金: {formatMoney(point.cash_balance)}
          </p>
          <p className="text-slate-300 text-sm">
            持仓: {formatMoney(point.position_value)}
          </p>
          {point.drawdown > 0 && (
            <p className="text-red-400 text-sm">
              回撤: -{point.drawdown.toFixed(2)}%
            </p>
          )}
        </div>
      );
    }
    return null;
  };

  return (

          data={peakPoints}

          {/* 初始资金参考线 */}
          <ReferenceLine
```

```tsx
import { TradePairDetail } from "./TradePairDetail";

interface TradePair {
  pair_id: string;
  symbol: string;
  side: "LONG" | "SHORT";
  quantity: number;
  entry_price: number;
  exit_price?: number;
  entry_time: string;
  exit_time?: string;
  holding_hours?: number;
  pnl?: number;
  pnl_pct?: number;
  status: "OPEN" | "CLOSED";
}

interface TradePairTableProps {
  pairs: TradePair[];
  loading?: boolean;
}

export function TradePairTable({ pairs, loading }: TradePairTableProps) {
  const [selectedPair, setSelectedPair] = useState<TradePair | null>(null);
  const [sortConfig, setSortConfig] = useState<{
    key: keyof TradePair;
    direction: "asc" | "desc";
  }>({ key: "entry_time", direction: "desc" });

  // 排序
  const sortedPairs = [...pairs].sort((a, b) => {
    const aVal = a[sortConfig.key];
    const bVal = b[sortConfig.key];
    
    if (aVal === undefined) return 1;
    if (bVal === undefined) return -1;
    
    if (aVal < bVal) return sortConfig.direction === "asc" ? -1 : 1;
    if (aVal > bVal) return sortConfig.direction === "asc" ? 1 : -1;
    return 0;
  });

  const handleSort = (key: keyof TradePair) => {
    setSortConfig({
      key,
      direction: sortConfig.key === key && sortConfig.direction === "asc" ? "desc" : "asc",
    });
  };

  const SortIcon = ({ columnKey }: { columnKey: keyof TradePair }) => {
    if (sortConfig.key !== columnKey) return null;
    return sortConfig.direction === "asc" ? (
      <ChevronUp className="inline w-4 h-4 ml-1" />
    ) : (
      <ChevronDown className="inline w-4 h-4 ml-1" />
    );
  };

  const formatHoldingTime = (hours?: number) => {
    if (!hours) return "-";
    if (hours < 1) return `${Math.round(hours * 60)}分钟`;
    if (hours < 24) return `${hours.toFixed(1)}小时`;
    return `${(hours / 24).toFixed(1)}天`;
  };

  if (loading) {
    return (
      <div className="flex items-center justify-center h-64">
        <div className="animate-spin rounded-full h-8 w-8 border-b-2 border-blue-500" />
      </div>
    );
  }

  return (
    <>
      <div className="rounded-lg border border-slate-700 overflow-hidden">
        <Table>
          <TableHeader className="bg-slate-800">
            <TableRow>
              <TableHead
                className="cursor-pointer hover:bg-slate-700"
                onClick={() => handleSort("symbol")}
              >
                交易对 <SortIcon columnKey="symbol" />
              </TableHead>
              <TableHead
                className="cursor-pointer hover:bg-slate-700"
                onClick={() => handleSort("side")}
              >
                方向
              </TableHead>
              <TableHead
                className="cursor-pointer hover:bg-slate-700 text-right"
                onClick={() => handleSort("quantity")}
              >
                数量
              </TableHead>
              <TableHead
                className="cursor-pointer hover:bg-slate-700 text-right"
                onClick={() => handleSort("entry_price")}
              >
                入场价
              </TableHead>
              <TableHead
                className="cursor-pointer hover:bg-slate-700 text-right"
                onClick={() => handleSort("exit_price")}
              >
                出场价
              </TableHead>
              <TableHead
                className="cursor-pointer hover:bg-slate-700"
                onClick={() => handleSort("entry_time")}
              >
                入场时间 <SortIcon columnKey="entry_time" />
              </TableHead>
              <TableHead
                className="cursor-pointer hover:bg-slate-700"
                onClick={() => handleSort("holding_hours")}
              >
                持仓时长
              </TableHead>
              <TableHead
                className="cursor-pointer hover:bg-slate-700 text-right"
                onClick={() => handleSort("pnl")}
              >
                盈亏 <SortIcon columnKey="pnl" />
              </TableHead>
              <TableHead
                className="cursor-pointer hover:bg-slate-700"
                onClick={() => handleSort("status")}
              >
                状态
              </TableHead>
              <TableHead>详情</TableHead>
            </TableRow>
          </TableHeader>
          <TableBody>
            {sortedPairs.map((pair) => (
              <TableRow
                key={pair.pair_id}
                className="hover:bg-slate-800/50 transition-colors"
              >
                <TableCell className="font-mono font-medium">
                  {pair.symbol}
                </TableCell>
                <TableCell>
                  <Badge
                    variant={pair.side === "LONG" ? "default" : "destructive"}
                    className={pair.side === "LONG" ? "bg-green-600" : ""}
                  >
                    {pair.side}
                  </Badge>
                </TableCell>
                <TableCell className="text-right font-mono">
                  {pair.quantity.toFixed(4)}
                </TableCell>
                <TableCell className="text-right font-mono">
                  {formatMoney(pair.entry_price)}
                </TableCell>
                <TableCell className="text-right font-mono">
                  {pair.exit_price ? formatMoney(pair.exit_price) : "-"}
                </TableCell>
                <TableCell className="text-slate-400 text-sm">
                  {formatDateTime(pair.entry_time)}
                </TableCell>
                <TableCell className="text-slate-300">
                  {formatHoldingTime(pair.holding_hours)}
                </TableCell>
                <TableCell className="text-right">
                  {pair.pnl !== undefined ? (
                    <>
                      <div

                        }
                      >
                        {pair.pnl >= 0 ? "+" : ""}
                        {formatMoney(pair.pnl)}
```

```tsx
"use client";

import { useEffect, useRef, useState, useCallback } from "react";

interface WebSocketHookOptions {
  url: string;
  onMessage?: (data: any) => void;
  onOpen?: () => void;
  onClose?: () => void;
  onError?: (error: Event) => void;
  reconnectInterval?: number;
  maxReconnectAttempts?: number;
}

export function useWebSocket({
  url,
  onMessage,
  onOpen,
  onClose,
  onError,
  reconnectInterval = 5000,
  maxReconnectAttempts = 10,
}: WebSocketHookOptions) {
  const [isConnected, setIsConnected] = useState(false);
  const [lastMessage, setLastMessage] = useState<any>(null);
  const wsRef = useRef<WebSocket | null>(null);
  const reconnectAttemptsRef = useRef(0);
  const reconnectTimeoutRef = useRef<NodeJS.Timeout | null>(null);

  const connect = useCallback(() => {
    try {
      if (wsRef.current?.readyState === WebSocket.OPEN) {
        return;
      }

      const ws = new WebSocket(url);
      wsRef.current = ws;

      ws.onopen = () => {
        console.log("WebSocket connected");
        setIsConnected(true);
        reconnectAttemptsRef.current = 0;
        onOpen?.();
      };

      ws.onmessage = (event) => {
        try {
          const data = JSON.parse(event.data);
          setLastMessage(data);
          onMessage?.(data);
        } catch (e) {
          console.error("Failed to parse WebSocket message:", e);
        }
      };

      ws.onclose = (event) => {
        console.log("WebSocket closed", event);
        setIsConnected(false);
        onClose?.();

        // 自动重连
        if (reconnectAttemptsRef.current < maxReconnectAttempts) {
          reconnectAttemptsRef.current++;
          console.log(
            `Attempting to reconnect (${reconnectAttemptsRef.current}/${maxReconnectAttempts})...`
          );
          reconnectTimeoutRef.current = setTimeout(connect, reconnectInterval);
        }
      };

      ws.onerror = (error) => {
        console.error("WebSocket error:", error);
        onError?.(error);
      };
    } catch (error) {
      console.error("Failed to connect WebSocket:", error);
    }
  }, [url, onOpen, onClose, onError, onMessage, reconnectInterval, maxReconnectAttempts]);

  const disconnect = useCallback(() => {
    if (reconnectTimeoutRef.current) {
      clearTimeout(reconnectTimeoutRef.current);
      reconnectTimeoutRef.current = null;
    }

    if (wsRef.current) {
      wsRef.current.close();
      wsRef.current = null;
    }
    setIsConnected(false);
  }, []);

  const send = useCallback((data: any) => {
    if (wsRef.current?.readyState === WebSocket.OPEN) {
      wsRef.current.send(JSON.stringify(data));
    } else {
      console.warn("WebSocket not connected, cannot send message");
    }
  }, []);

  useEffect(() => {
    connect();
    return () => disconnect();
  }, [connect, disconnect]);

  return {
    isConnected,
    lastMessage,
    send,
    connect,
    disconnect,
  };
}
```

## 八、交易成本分析（TCA）核心模块补充

### 8.1 TCA 核心概念与价值

交易成本分析（Transaction Cost Analysis）是量化交易系统的关键组成部分，用于：
- **执行质量评估**：衡量订单执行效率，识别可优化环节
- **成本归因**：分解交易成本构成（冲击成本、滑点、手续费等）
- **算法交易对比**：VWAP/TWAP等算法的绩效评估
- **策略改进**：基于TCA数据优化下单策略和执行参数

### 8.2 执行缺口（Implementation Shortfall）分析

#### 8.2.1 理论框架

**Perold's Model**（行业标准）：
```
IS = Paper Portfolio Return - Real Portfolio Return
```

**分解公式**：
```
IS = Delay Cost + Trading Cost + Opportunity Cost

其中：
- Delay Cost = (到达价格 - 决策价格) × 交易量
- Trading Cost = (执行价格 - 到达价格) × 交易量
- Opportunity Cost = (收盘价格 - 执行价格) × (计划量 - 成交量)
```

#### 8.2.3 执行缺口计算服务

```python
# backend/app/services/tca_service.py

from datetime import datetime, timedelta
from decimal import Decimal
from typing import Dict, List, Optional
from collections import defaultdict

class TCAService:
    """交易成本分析服务"""
    
    async def calculate_implementation_shortfall(
        self,
        trade: PaperTrade,
        decision_price: Optional[Decimal] = None,
        decision_time: Optional[datetime] = None,
        planned_qty: Optional[Decimal] = None
    ) -> Dict:
        """
        计算执行缺口（Implementation Shortfall）
        
        Perold模型：IS = 纸面组合收益 - 真实组合收益
        分解为：延迟成本 + 交易成本 + 机会成本
        """
        
        # 获取价格基准
        arrival_price = await self._get_arrival_price(
            trade.symbol, 
            trade.created_at
        )
        
        close_price = await self._get_close_price(
            trade.symbol,
            trade.created_at.date()
        )
        
        # 决策价格默认为订单创建前的最新价格
        if not decision_price:
            decision_price = await self._get_decision_price(
                trade.symbol,
                trade.created_at
            )
        
        if not decision_time:
            decision_time = trade.created_at - timedelta(seconds=1)
        
        executed_qty = Decimal(str(trade.quantity))
        execution_price = Decimal(str(trade.price))
        
        if not planned_qty:
            planned_qty = executed_qty
        
        # 计算各部分成本
        delay_cost = self._calculate_delay_cost(
            decision_price,
            arrival_price,
            executed_qty,
            trade.side
        )
        
        trading_cost = self._calculate_trading_cost(
            arrival_price,
            execution_price,
            executed_qty,
            trade.side
        )
        
        opportunity_cost = self._calculate_opportunity_cost(
            execution_price,
            close_price,
            planned_qty,
            executed_qty,
            trade.side
        )
        
        total_is = delay_cost + trading_cost + opportunity_cost
        
        # 计算基点（相对于交易价值）
        trade_value = execution_price * executed_qty
        is_bps = (total_is / trade_value * 10000) if trade_value > 0 else Decimal('0')
        
        return {
            "trade_id": trade.id,
            "symbol": trade.symbol,
            "side": trade.side,
            
            "decision_price": float(decision_price),
            "arrival_price": float(arrival_price),
            "execution_price": float(execution_price),
            "close_price": float(close_price) if close_price else None,
            
            "planned_qty": float(planned_qty),
            "executed_qty": float(executed_qty),
            
            "delay_cost": float(delay_cost),
            "trading_cost": float(trading_cost),
            "opportunity_cost": float(opportunity_cost),
            "total_is": float(total_is),
            "is_bps": float(is_bps),
            
            "decision_time": decision_time.isoformat(),
            "arrival_time": trade.created_at.isoformat(),
            "execution_time": trade.created_at.isoformat()
        }
    
    def _calculate_delay_cost(
        self,
        decision_price: Decimal,
        arrival_price: Decimal,
        qty: Decimal,
        side: str
    ) -> Decimal:
        """延迟成本：从决策到订单到达市场期间的价格变动"""
        if side == "BUY":
            return (arrival_price - decision_price) * qty
        else:  # SELL
            return (decision_price - arrival_price) * qty
    
    def _calculate_trading_cost(
        self,
        arrival_price: Decimal,
        execution_price: Decimal,
        qty: Decimal,
        side: str
    ) -> Decimal:
        """交易成本：订单执行相对于到达价格的成本（含冲击成本）"""
        if side == "BUY":
            return (execution_price - arrival_price) * qty
        else:  # SELL
            return (arrival_price - execution_price) * qty
    
    def _calculate_opportunity_cost(
        self,
        execution_price: Decimal,
        close_price: Optional[Decimal],
        planned_qty: Decimal,
        executed_qty: Decimal,
        side: str
    ) -> Decimal:
        """机会成本：未成交量在收盘时的机会成本"""
        if not close_price:
            return Decimal('0')
        
        unexecuted_qty = planned_qty - executed_qty
        if unexecuted_qty <= 0:
            return Decimal('0')
        
        if side == "BUY":
            return (close_price - execution_price) * unexecuted_qty
        else:  # SELL
            return (execution_price - close_price) * unexecuted_qty
    
    async def calculate_vwap_comparison(
        self,
        trade: PaperTrade,
        interval: str = "1h"
    ) -> Dict:
        """
        计算与VWAP基准的对比
        
        VWAP偏差 = (执行价格 - 区间VWAP) / 区间VWAP × 10000 (bps)
        """
        execution_price = Decimal(str(trade.price))
        qty = Decimal(str(trade.quantity))
        
        # 获取交易时间窗口的VWAP
        vwap = await self._get_interval_vwap(
            trade.symbol,
            trade.created_at,
            interval
        )
        
        if not vwap:
            return None
        
        # 计算偏差
        if trade.side == "BUY":
            vwap_slippage_bps = ((execution_price - vwap) / vwap * 10000)
        else:  # SELL
            vwap_slippage_bps = ((vwap - execution_price) / vwap * 10000)
        
        # 计算相对于VWAP的成本
        if trade.side == "BUY":
            vwap_cost = (execution_price - vwap) * qty
        else:
            vwap_cost = (vwap - execution_price) * qty
        
        return {
            "trade_id": trade.id,
            "symbol": trade.symbol,
            "side": trade.side,
            "execution_price": float(execution_price),
            "vwap": float(vwap),
            "vwap_slippage_bps": float(vwap_slippage_bps),
            "vwap_cost": float(vwap_cost),
            "interval": interval
        }
    
    async def get_tca_summary(
        self,
        start_date: datetime,
        end_date: datetime,
        symbol: Optional[str] = None
    ) -> Dict:
        """获取TCA汇总报告"""
        
        tca_records = await self._get_tca_records(start_date, end_date, symbol)
        
        if not tca_records:
            return None
        
        # 汇总统计
        total_trades = len(tca_records)
        total_is = sum(r.total_is for r in tca_records)
        avg_is_bps = sum(r.is_bps for r in tca_records) / total_trades
        
        # 成本分解
        avg_delay_cost = sum(r.delay_cost for r in tca_records) / total_trades
        avg_trading_cost = sum(r.trading_cost for r in tca_records) / total_trades
        avg_opportunity_cost = sum(r.opportunity_cost for r in tca_records) / total_trades
        
        # VWAP对比统计
        vwap_records = [r for r in tca_records if r.vwap_slippage_bps is not None]
        if vwap_records:
            avg_vwap_slippage = sum(r.vwap_slippage_bps for r in vwap_records) / len(vwap_records)
            vwap_win_rate = sum(1 for r in vwap_records if r.vwap_slippage_bps <= 0) / len(vwap_records) * 100
        else:
            avg_vwap_slippage = 0
            vwap_win_rate = 0
        
        return {
            "period": {
                "start": start_date.isoformat(),
                "end": end_date.isoformat()
            },
            "summary": {
                "total_trades": total_trades,
                "total_is": float(total_is),
                "avg_is_bps": float(avg_is_bps)
            },
            "cost_breakdown": {
                "avg_delay_cost": float(avg_delay_cost),
                "avg_trading_cost": float(avg_trading_cost),
                "avg_opportunity_cost": float(avg_opportunity_cost)
            },
            "vwap_comparison": {
                "avg_vwap_slippage_bps": float(avg_vwap_slippage),
                "vwap_win_rate": float(vwap_win_rate)
            }
        }
```

### 8.3 滑点分析（Slippage Analysis）

#### 8.3.1 滑点定义与分类

**滑点** = 订单预期价格 - 实际执行价格

**滑点分类**：
- **正向滑点**：实际价格优于预期（盈利）
- **负向滑点**：实际价格劣于预期（亏损）

**滑点来源**：
1. 买卖价差（Bid-Ask Spread）
2. 市场冲击成本（Market Impact）
3. 延迟滑点（Latency Slippage）
4. 波动率滑点（Volatility Slippage）

#### 8.3.2 滑点分析服务

```python
async def analyze_slippage(
    self,
    trades: List[PaperTrade],
    window_minutes: int = 5
) -> Dict:
    """
    综合滑点分析
    
    参数:
        trades: 交易记录列表
        window_minutes: 计算基准价格的时间窗口
    """
    
    slippage_stats = defaultdict(list)
    
    for trade in trades:
        # 获取订单创建时的买卖盘
        book_snapshot = await self._get_order_book_snapshot(
            trade.symbol,
            trade.created_at
        )
        
        if not book_snapshot:
            continue
        
        # 根据方向确定基准价格
        if trade.side == "BUY":
            expected_price = book_snapshot['ask_price']  # 以卖一价为预期
        else:
            expected_price = book_snapshot['bid_price']  # 以买一价为预期
        
        execution_price = Decimal(str(trade.price))
        qty = Decimal(str(trade.quantity))
        
        # 计算滑点
        if trade.side == "BUY":
            slippage = expected_price - execution_price
            slippage_bps = (slippage / expected_price * 10000) if expected_price > 0 else Decimal('0')
        else:
            slippage = execution_price - expected_price
            slippage_bps = (slippage / expected_price * 10000) if expected_price > 0 else Decimal('0')
        
        slippage_amount = slippage * qty
        
        slippage_stats[trade.symbol].append({
            "trade_id": trade.id,
            "side": trade.side,
            "expected_price": float(expected_price),
            "execution_price": float(execution_price),
            "slippage": float(slippage),
            "slippage_bps": float(slippage_bps),
            "slippage_amount": float(slippage_amount),
            "spread": float(book_snapshot.get('spread', 0)),
            "spread_bps": float(book_snapshot.get('spread_bps', 0))
        })
    
    # 汇总统计
    summary = {}
    all_slippage_bps = []
    
    for symbol, records in slippage_stats.items():
        symbol_bps = [r['slippage_bps'] for r in records]
        all_slippage_bps.extend(symbol_bps)
        
        positive = sum(1 for r in records if r['slippage_bps'] >= 0)
        negative = sum(1 for r in records if r['slippage_bps'] < 0)
        
        summary[symbol] = {
            "total_trades": len(records),
            "avg_slippage_bps": sum(symbol_bps) / len(records),
            "median_slippage_bps": sorted(symbol_bps)[len(records) // 2],
            "positive_slippage_rate": positive / len(records) * 100,
            "negative_slippage_rate": negative / len(records) * 100,
            "total_slippage_amount": sum(r['slippage_amount'] for r in records)
        }
    
    # 全市场汇总
    if all_slippage_bps:
        overall = {
            "total_trades": len(all_slippage_bps),
            "avg_slippage_bps": sum(all_slippage_bps) / len(all_slippage_bps),
            "median_slippage_bps": sorted(all_slippage_bps)[len(all_slippage_bps) // 2],
            "p95_slippage_bps": sorted(all_slippage_bps)[int(len(all_slippage_bps) * 0.95)],
            "p99_slippage_bps": sorted(all_slippage_bps)[int(len(all_slippage_bps) * 0.99)]
        }
    else:
        overall = None
    
    return {
        "overall": overall,
        "by_symbol": summary
    }
```

### 8.4 VWAP / TWAP 算法交易绩效评估

#### 8.4.1 VWAP 基准计算

```python
async def calculate_vwap_benchmark(
    self,
    symbol: str,
    start_time: datetime,
    end_time: datetime,
    interval: str = "1m"
) -> List[Dict]:
    """
    计算VWAP基准曲线
    
    VWAP = Σ(Price × Volume) / Σ(Volume)
    """
    
    klines = await self._get_klines(symbol, start_time, end_time, interval)
    
    if not klines:
        return []
    
    cumulative_volume = Decimal('0')
    cumulative_value = Decimal('0')
    vwap_series = []
    
    for kline in klines:
        # 使用典型价格 (High + Low + Close) / 3
        typical_price = (Decimal(str(kline['high'])) + 
                        Decimal(str(kline['low'])) + 
                        Decimal(str(kline['close']))) / Decimal('3')
        
        volume = Decimal(str(kline['volume']))
        
        cumulative_volume += volume
        cumulative_value += typical_price * volume
        
        if cumulative_volume > 0:
            vwap = cumulative_value / cumulative_volume
        else:
            vwap = typical_price
        
        vwap_series.append({
            "timestamp": kline['timestamp'],
            "open": float(kline['open']),
            "high": float(kline['high']),
            "low": float(kline['low']),
            "close": float(kline['close']),
            "volume": float(volume),
            "typical_price": float(typical_price),
            "vwap": float(vwap),
            "cumulative_volume": float(cumulative_volume)
        })
    
    return vwap_series
```

#### 8.4.2 算法绩效指标

| 指标 | 说明 | 目标 |
|------|------|------|
| **VWAP偏差** | (执行均价 - VWAP) / VWAP × 10000 (bps) | 越小越好（负值为优） |
| **参与率** | 成交量 / 市场同期成交量 | 通常 5-20% |
| **实施短fall** | 相对于到达价格的总成本 | 越小越好 |
| **成交率** | 成交量 / 计划量 | 越高越好（视策略而定） |
